De Auto

Dit is de auto in de game die rijd die ook samen werkt met de wheel WheelCollider. De gif die er staat laat zien hoe de auto rijd, en met de Instellingen van de wheel collider zo dat je ook echt een gevoel krijgt hoe een auto rijdt.

In de code laat ik zien hoe de wheel collider in zijn werk gaat. 


public class CarSelection : MonoBehaviour
{
    public GameObject[] cars;

    public Button next;

    public Button prev;

    int index;

    public void Race()
    {
        SceneManager.LoadSceneAsync("Main");
    }

    void Start()
    {
        index = PlayerPrefs.GetInt("carsIndex");

        for(int i = 0; i < cars.Length; i++)
        {
            cars[i].SetActive(false);
            cars[index].SetActive(true);
        }

        if (index < cars.Length)
        {
            cars[index].SetActive(true);
        }
        else
        {
            Debug.LogError("Index out of bounds: " + index);
        }
    }

    void Update()
    {
        if(index >= 2)
        {
            next.interactable = false;
        }
        else
        {
            next.interactable = true;
        }

        if (index <= 1)
        {
            prev.interactable = false;
        }
        else
        {
            prev.interactable = true;
        }
    }

    public void Next()
    {
        index++;

        if (index >= cars.Length)
        {
            index = 0;
        }

        SetActiveCar();
        PlayerPrefs.SetInt("carsIndex", index);
        PlayerPrefs.Save();
    }

    public void Prev()
    {
        index--;

        if (index < 0)
        {
            index = cars.Length - 1;
        }

        SetActiveCar();
        PlayerPrefs.SetInt("carsIndex", index);
        PlayerPrefs.Save();
    }

    private void SetActiveCar()
    {
        for (int i = 0; i < cars.Length; i++)
        {
            cars[i].SetActive(i == index);
        }
    }
}

De wielen van de auto

Dit zijn de wheel die op de wielen zitten zo dat je ook een mooi effect krijgt van ook echt rollende wielen. 


public class Wheel : MonoBehaviour
{
    public bool IsPowered;
    public bool CanTurn;

    public GameObject Visualls;

    private WheelCollider wheelCollider;

    public void Start()
    {
        wheelCollider = GetComponent();
    }
    private void Update()
    {
        Vector3 Position;
        Quaternion Rotation;
        wheelCollider.GetWorldPose(out Position, out Rotation);
        Visualls.transform.position = Position;
        Visualls.transform.rotation = Rotation;
    }
    public void Accelerate(float power)
    {
        if (IsPowered)
        {
            wheelCollider.motorTorque = power;
        }
    }
    public void Turn(float Angle)
    {
        if(CanTurn)
        {
            wheelCollider.steerAngle = Angle;
        }
    }
    public void Brake(float power)
    {
        wheelCollider.brakeTorque = power;
    }
}

Het selecteren van de auto

Dit is de selectie van de auto's die je uit de game kon kiezen. 


public class CarSelection : MonoBehaviour
{
    public GameObject[] cars;

    public Button next;

    public Button prev;

    int index;
    public void Race()
    {
        SceneManager.LoadSceneAsync("Main");
    }

    void Start()
    {
        index = PlayerPrefs.GetInt("carsIndex");

        for(int i = 0; i < cars.Length; i++)
        {
            cars[i].SetActive(false);
            cars[index].SetActive(true);
        }

        if (index < cars.Length)
        {
            // Access the array element
            cars[index].SetActive(true);
        }
        else
        {
            // Handle the case where the index is out of bounds
            Debug.LogError("Index out of bounds: " + index);
        }
    }

    void Update()
    {
        if(index >= 2)
        {
            next.interactable = false;
        }
        else
        {
            next.interactable = true;
        }

        if (index <= 1)
        {
            prev.interactable = false;
        }
        else
        {
            prev.interactable = true;
        }
    }

    public void Next()
    {
        // Increment the index
        index++;

        // Check if the index is within the bounds of the array
        if (index >= cars.Length)
        {
            // If index exceeds the array length, wrap around to the beginning
            index = 0;
        }

        // Activate the car corresponding to the updated index
        SetActiveCar();

        // Save the updated index
        PlayerPrefs.SetInt("carsIndex", index);
        PlayerPrefs.Save();
    }

    public void Prev()
    {
        // Decrement the index
        index--;

        // Check if the index is within the bounds of the array
        if (index < 0)
        {
            // If index becomes negative, wrap around to the end of the array
            index = cars.Length - 1;
        }

        // Activate the car corresponding to the updated index
        SetActiveCar();

        // Save the updated index
        PlayerPrefs.SetInt("carsIndex", index);
        PlayerPrefs.Save();
    }

    // Helper method to activate the car corresponding to the current index
    private void SetActiveCar()
    {
        for (int i = 0; i < cars.Length; i++)
        {
            cars[i].SetActive(i == index);
        }
    }
}
